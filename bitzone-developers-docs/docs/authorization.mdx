import React, { useState } from 'react';

export const SESSION_COOKIE = 'DOCS_SESSION';
export const SESSION_VALUE = 'ok';

export function logout() {
  try { document.cookie = SESSION_COOKIE + '=; Path=/; Max-Age=0'; } catch (e) {}
  const base = (window.__docusaurus && window.__docusaurus.baseUrl) || '/TechDoc/';
  const b = base.endsWith('/') ? base : base + '/';
  window.location.replace(window.location.origin + b + 'authorization');
}

export function AuthForm() {
  const [username, setUsername] = useState('');
  const [password, setPassword] = useState('');
  const [error, setError] = useState(null);

  function onSubmit(e) {
    e.preventDefault();
    if (username === 'admin' && password === 'admin') {
      document.cookie = SESSION_COOKIE + '=' + SESSION_VALUE + '; SameSite=Lax; Path=/; Max-Age=86400';
      window.location.replace((window.__docusaurus?.baseUrl || '/TechDoc/') );
      return;
    }
    setError('Неверный логин или пароль');
  }

  return (
    <form onSubmit={onSubmit} className="auth-card">
      <h2>Авторизация</h2>
      {error ? <p className="auth-error">{error}</p> : null}
      <label htmlFor="username">Логин</label>
      <input id="username" value={username} onChange={(e)=>setUsername(e.target.value)} required />
      <label htmlFor="password">Пароль</label>
      <input id="password" type="password" value={password} onChange={(e)=>setPassword(e.target.value)} required />
      <button type="submit">Войти</button>
      <button type="button" className="btn-logout" onClick={logout}>Выйти</button>
      <div className="auth-hint">По умолчанию: admin / admin</div>
    </form>
  );
}
